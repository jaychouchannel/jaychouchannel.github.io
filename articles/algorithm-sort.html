深入理解 JavaScript 排序算法：从原理到实现

Jaychouchannel · 2023-06-15

排序是计算机科学中的基础问题，本篇覆盖冒泡、快速、归并、堆排序等常见算法，并给出 JavaScript 示例。

冒泡排序（Bubble Sort）

// 冒泡排序
function bubbleSort(arr) {
  const a = arr.slice();
  for (let i = 0; i < a.length; i++) {
    for (let j = 0; j < a.length - 1 - i; j++) {
      if (a[j] > a[j + 1]) [a[j], a[j + 1]] = [a[j + 1], a[j]];
    }
  }
  return a;
}

快速排序（Quick Sort）

// 快速排序（递归）
function quickSort(arr) {
  if (arr.length <= 1) return arr.slice();
  const a = arr.slice();
  const pivot = a[Math.floor(a.length / 2)];
  const left = [], right = [], eq = [];
  a.forEach(v => {
    if (v < pivot) left.push(v);
    else if (v > pivot) right.push(v);
    else eq.push(v);
  });
  return quickSort(left).concat(eq, quickSort(right));
}
